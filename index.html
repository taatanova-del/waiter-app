  <h2>Меню - Стол <span id="currentTable"></span></h2>

  <div id="menu-items"></div>

<script>

let currentTable;

const menuItems = [

  { name: 'Салат', price: 5 },

  { name: 'Суп', price: 3 },

  { name: 'Стейк', price: 10 },

  { name: 'Напиток', price: 2 },

  { name: 'Десерт', price: 4 }

];

let order = [];


// Логин (для демонстрации любой ввод проходит)

function login() {

  const username = document.getElementById('username').value;

  const password = document.getElementById('password').value;

  if(username && password) {

    document.getElementById('login').classList.remove('active');

    document.getElementById('tables').classList.add('active');

  } else alert('Введите имя и пароль');

}


// Выбор стола

function openTable(table){

  currentTable = table;

  document.getElementById('currentTable').innerText = table;

  document.getElementById('tables').classList.remove('active');

  document.getElementById('menu').classList.add('active');

  renderMenu();

}


// Рендер меню

function renderMenu(){

  const container = document.getElementById('menu-items');

  container.innerHTML = '';

  menuItems.forEach(item=>{

    const div = document.createElement('div');

    div.className = 'menu-item';

    div.innerHTML = `

      <span>${item.name} - $${item.price}</span>

      <div class="counter">

        <button onclick="updateOrder('${item.name}', -1)">−</button>

        <input type="number" value="0" min="0" id="count-${item.name}" readonly>

        <button onclick="updateOrder('${item.name}', 1)">+</button>

      </div>

    `;

    container.appendChild(div);

  });

}


// Изменение количества

function updateOrder(itemName, delta){

  const input = document.getElementById(`count-${itemName}`);

  let val = parseInt(input.value) + delta;

  if(val < 0) val = 0;

  input.value = val;


  const index = order.findIndex(o=>o.item===itemName);

  if(index >= 0) {

    order[index].quantity = val;

  } else {

    order.push({item: itemName, quantity: val});

  }

  // Фильтруем нули

  order = order.filter(o=>o.quantity>0);

}


// Закрытие заказа

function closeOrder(){

  if(order.length===0){ alert('Добавьте хотя бы одну позицию'); return; }

  if(!confirm('Закрыть заказ?')) return;


  console.log(`Заказ со стола ${currentTable}:`, order);

  alert('Заказ сохранён (сейчас только в консоли браузера)');


  order = [];

  // Сброс счётчиков

  menuItems.forEach(item=>{

    document.getElementById(`count-${item.name}`).value = 0;

  });

  document.getElementById('menu').classList.remove('active');

  document.getElementById('tables').classList.add('active');

}
